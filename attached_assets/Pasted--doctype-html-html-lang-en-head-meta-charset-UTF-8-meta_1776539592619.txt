    function setNav(label) {
      document.getElementById('navMini').textContent = label;
    }

    function go(screen) {
      document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
      document.getElementById(screen).classList.add('active');
      const labelMap = {
        landing: 'Landing',
        top5: 'Top 5',
        chart: 'Live Chart',
        decision: 'Trade Decision',
        watchlist: 'Watchlist',
        premium: 'Premium'
      };
      setNav(labelMap[screen] || 'App');
      if (screen === 'top5') renderTop5();
      if (screen === 'chart') renderChart();
      if (screen === 'decision') renderDecision();
      if (screen === 'watchlist') renderWatchlist();
    }

    function renderTop5() {
      const el = document.getElementById('top5List');
      el.innerHTML = state.top5.map(t => `
        <div class="top5-card" onclick="state.selected='${t.ticker}'; go('chart')">
          <div>
            <div class="eyebrow">${t.ticker}</div>
            <h3 class="${t.side === 'BUY' ? 'buy' : 'sell'}">${t.side}</h3>
            <div class="note">${t.reason}</div>
          </div>
          <div class="score">${t.score}</div>
        </div>
      `).join('');
    }

    function renderChart() {
      const inst = state.watchlist.find(x => x.ticker === state.selected) || state.watchlist[0];
      document.getElementById('chartTicker').textContent = inst.ticker;
      document.getElementById('chartMeta').textContent = `${inst.cat} · Live Chart`;
    }

    function renderDecision() {
      const d = state.decision[state.selected] || state.decision.GOLD;
      document.getElementById('decisionTitle').textContent = `${d.side} ${state.selected}`;
      document.getElementById('decisionConfidence').textContent = d.confidence;
      document.getElementById('entry').textContent = d.entry;
      document.getElementById('stop').textContent = d.stop;
      document.getElementById('tp1').textContent = d.tp1;
      document.getElementById('tp2').textContent = d.tp2;
      const bias = document.getElementById('decisionBias');
      bias.textContent = d.bias;
      bias.className = `bias-badge ${d.bias === 'Risk-On' ? 'bias-on' : 'bias-off'}`;
      document.getElementById('decisionWhy').textContent = d.why;
    }

    function renderWatchlist() {
      const market = state.watchlist.filter(x => x.cat !== 'Crypto');
      const crypto = state.watchlist.filter(x => x.cat === 'Crypto');
      const row = x => `
        <div class="watch-row">
          <div>
            <strong>${x.ticker}</strong>
            <div class="note">${x.cat}</div>
          </div>
          <div class="chips">
            <span class="chip ${x.side === 'BUY' ? 'buy' : 'sell'}">${x.side}</span>
            <span class="chip">${x.trend}</span>
            <button class="btn" onclick="state.selected='${x.ticker}'; go('chart')">Open</button>
          </div>
        </div>
      `;
      document.getElementById('marketList').innerHTML = market.map(row).join('');
      document.getElementById('cryptoList').innerHTML = crypto.map(row).join('');
    }

    renderTop5();
